    class Test_integrity(unittest.TestCase):
    """
    Tests the integrity of each table in the cycling db.
    "Integrity" in this case is the proportion of values that are null or empty
    compared to the total number of rows in a given table.
    eg: .90 means 90% of data for a given column is valid--meaning not null or empty
    """
    def test_race_info(self, high_threshold: float = .90, medium_threshold: float = .70, low_threshold: float = .60):
        """
        Tests the [race_info](../database-schema/#race_info) table by ensuring that each field
        returns at least `threshold` % valid values, i.e. not `None` or `""`

        Args:
            high_threshold: Percent (expressed as a decimal) of each field in `high_integrity_fields` that must be valid.
            medium_threshold: Percent (expressed as a decimal) of each field in `medium_integrity_fields` that must be valid.
            low_threshold: Percent (expressed as a decimal) of each field in `low_integrity_fields` that must be valid.
        """
        builder = queryBuilder()
        cursor = builder.cycling_db.cursor()
        testQuery = """SELECT * FROM race_info"""
        cursor.execute(testQuery)
        res = list(cursor.fetchall())
        cursor.close()
        formatted_data = []
        for row in res:
            formatted_data.append(
                {
                    'race_url': row[0],
                    'is_total_results_page': row[1],
                    'class': row[2],
                    'parcour_type': row[3],
                    'date': row[4],
                    'distance': row[5],
                    'profile_score': row[6],
                    'vertical_meters': row[7],
                    'race_ranking': row[8],
                    'startlist_quality_score': row[9]
                }
            )
        integrity = dbUpdater.check_integrity(formatted_data)
        # these fields typically have very high integrity
        high_integrity_fields = ['race_url', 'is_total_results_page', 'class', 'parcour_type', 'Date', 'Distance']
        # these fields typically have medium integrity
        medium_integrity_fields = ['Race ranking', 'Startlist quality score']
        # these fields typically have low integrity
        low_integrity_fields = ['ProfileScore', 'Vertical meters']
        for key, value in integrity.items():
            print(f"{key}: {value}")
            if key in high_integrity_fields:
                self.assertGreaterEqual(value, high_threshold)
            if key in medium_integrity_fields:
                self.assertGreaterEqual(value, medium_threshold)
            if key in low_integrity_fields:
                self.assertGreaterEqual(value, low_threshold)
