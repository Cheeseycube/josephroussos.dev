    def test_bigBuilder_1(self):
        """
        Select all riders aged 23 or less who have finished in the top ten in 1.1 races
        with profile scores of 50 or greater
        in the past 4 years
        """
        # getting the builder's results for the given filters
        curr_year = datetime.today().year
        query_builder = queryBuilder()
        builderRes = query_builder.bigBuilder(
            [
                {'table': 'race_info', 'column': 'year(date)', 'compare': '>=', 'value': str(curr_year - 4)},
                {'table': 'race_info', 'column': 'year(date)', 'compare': '<=', 'value': str(curr_year)},
                {'table': 'race_results', 'column': 'placement', 'compare': '<=', 'value': '10'},
                {'table': 'rider_info', 'column': 'age', 'compare': '<=', 'value': '23'},
                {'table': 'race_info', 'column': 'profile_score', 'compare': '>=', 'value': '50'},
                {'table': 'race_info', 'column': 'class', 'compare': '=', 'value': '1.1'}
            ]
        )
        builderResList = [row[0] for row in builderRes]

        # getting the results for an equivalent sql query
        testQuery = f"""with
            selected_races as (select race_url from race_info
                                    where class  = "1.1" and profile_score >= 50
                                    and year(date) >= {curr_year - 4} and year(date) <= {curr_year}),
            selected_results as (select rider_url from race_results
                                    where race_url in (select selected_races.race_url from selected_races)
                                        and placement <= 10)
        select * from rider_info where rider_url in (select rider_url from selected_results)
                                    and age <= 23;"""
        cursor = query_builder.cycling_db.cursor()
        cursor.execute(testQuery)
        testQueryRes = cursor.fetchall()
        testQueryResList = [row[0] for row in testQueryRes]

        # comparing the differences
        self.maxDiff = None
        self.assertCountEqual(builderResList, testQueryResList)
